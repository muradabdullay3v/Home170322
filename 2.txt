using System;

class HT2
{

  static void Main ()
  {
Console.WriteLine(Reverse("Murad Salam Aye"));
  }

  static string Reverse (string str)
  {
    string rev = "";
    foreach(var word in str.Split(' ')){
        string strword = "";
        foreach(var chr in word){
            strword = chr + strword;
        }
        rev = rev + strword + " ";
    }
    return rev;
  }

}
