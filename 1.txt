using System;

class HT1
{

  static void Main ()
  {
    Console.WriteLine("Index value: " + IndexOf("Salam",'m',0,4));
  }

  static int IndexOf (string word,char simvol)
  {
    for (int i = 0; i < word.Length; i++)
      {
         if(word[i] == simvol){
             return i;
         }
      }
      return -1;
  }
  
   static int IndexOf (string word,char simvol,int index)
  {
    for (int i = index; i < word.Length; i++)
      {
         if(word[i] == simvol){
             return i;
         }
      }
      return -1;
  }
  
     static int IndexOf (string word,char simvol,int index,int count)
  {
    for (int i = index; i <= index + count; i++)
      {
         if(word[i] == simvol){
             return i;
         }
      }
      return -1;
  }

}
